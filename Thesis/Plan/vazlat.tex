\begin{enumerate}
  \item \textbf{Bevezetés}

  \item \textbf{Az emulátorokról}: Definíció, mûködés, fajtái, jelene, jövõje, jelentõsége

  \item \textbf{Nintendo Game Boy}
  \begin{enumerate}
    \item Történelme, hardverspecifikáció
    \item Architektúra
  \end{enumerate}

  \item \textbf{Emulátor fejlesztési folyamat/roadmap}: kutatási/implementálási/fejlesztési sorrend, az emulátor mûködése "madártávlatból"

  \item \textbf{Implementáció}

  \begin{enumerate}
    \item \textbf{Eszközök}
      \begin{enumerate}
        \item Rust nyelv, minifb library
        \item Források, dokumentációk, közösségi repozitórium, BGB Game Boy emulátor debuggere
      \end{enumerate}

    \item \textbf{Specifikáció}: input, output, elvárt mûködés
    \begin{enumerate}
      \item ROMok struktúrája, fajtáik
    \end{enumerate}

    \item \textbf{CPU}
      \begin{enumerate}
        \item Regiszterek, változók, flagek
        \item Órajelciklus, mûveletigény
      \end{enumerate}

    \item \textbf{Utasításkészlet}
      \begin{enumerate}
        \item Load (LD) utasítások
        \item Aritmetikus (ADD, ADC, SUB, SBC, INC, DEC) utasítások
        \item Logikai (AND, XOR, OR, CP) utasítások
        \item Jump, Call, Return (JP, CALL, RET) utasítások
        \item Verem utasítások (PUSH, POP)
        \item Bitmûveleti (RLC, RRC, RL, RR, SLA, SRA, SWAP, SRL, BIT, RES, SET) utasítások
        \item Egyéb, speciális utasítások
      \end{enumerate}

    \item \textbf{RAM/MMU}
      \begin{enumerate}
        \item Memóriatérkép, fontosabb, kiemelt helyek, adatstruktúra
        \item DMA
        \item Implementációja, trap-ek
        \item MBC
      \end{enumerate}

    \item \textbf{Idõzítõk}

    \item \textbf{LCD}
      \begin{enumerate}
        \item V-blank, H-blank, interrupt
      \end{enumerate}

    \item \textbf{PPU}
    \begin{enumerate}
      \item Alapvetõ mûködés, memóriaterületek
      \item Tile rendering
      \item Sprite rendering
      \begin{enumerate}
        \item Paletta, sprite attribútumok (flip, priority bit)
        \item Sprite Attribute Table
      \end{enumerate}
    \end{enumerate}

    \item \textbf{Boot Rom}: futtatás, mûködés

    \item \textbf{Joypad}
      \begin{enumerate}
        \item Kiosztás, wiring, mechanizmus
        \item Implementáció, jellemzõk, interrupt
      \end{enumerate}
  \end{enumerate}

  \item \textbf{Mûködés felhasználói szemmel}
  \begin{enumerate}
    \item Kinézet, screenshotok, használat
    \item Kapcsolók, további funkciók
    \item Dependenciák
  \end{enumerate}

  \item \textbf{Tesztelés}
  \begin{enumerate}
    \item Közösségi teszt ROMok, melyik mit tartalmaz
    \item A tesztelések eredményei, értékelés
  \end{enumerate}

  \item \textbf{Végszó}: mi nem került implementálásra, mit kellett volna másképp, a projekt jövõje
  \item Köszönetnyilvánítás
  \item Mellékletek


\end{enumerate}
